24 void CGA_Screen::setpos (unsigned short x, unsigned short y) {
25  // check whether the input is legit
26  if(x < TERMINAL_WIDTH && y < TERMINAL_HEIGHT){
27  // get IOPort access
28  IO_Port index_reg_cga(0x3D4);
29 
30  // the index register says where the data byte goes, which is why
31  // we have to write to the data register twice
32  IO_Port data_reg_cga(0x3D5);
33 
34  // calculate offset
35  unsigned short offset = x + (y * TERMINAL_WIDTH);
36 
37  unsigned char lowByte = offset & 0xFF;
38  unsigned char highByte = (offset >> 8) & 0xFF;
39 
40  // write index register for high byte
41  index_reg_cga.outb(CURSOR_POSITION_INDEX_HIGH);
42  // write new cursor high byte
43  data_reg_cga.outb(highByte);
44 
45  // write index register for high byte
46  index_reg_cga.outb(CURSOR_POSITION_INDEX_LOW);
47  // write new cursor high byte
48  data_reg_cga.outb(lowByte);
49  }
50 }

52 void CGA_Screen::getpos (unsigned short& x, unsigned short& y) const{
53  // get IOPort access
54  IO_Port index_reg_cga(0x3D4);
55  IO_Port data_reg_cga(0x3D5);
56 
57  // write index register for high byte
58  index_reg_cga.outb(CURSOR_POSITION_INDEX_HIGH);
59  // read high byte
60  unsigned char high_byte = data_reg_cga.inb();
61 
62  // write index register for low byte
63  index_reg_cga.outb(CURSOR_POSITION_INDEX_LOW);
64  // read low byte
65  unsigned char low_byte = data_reg_cga.inb();
66 
67  // create short from high and low byte
68  unsigned short offset = low_byte | (high_byte << 8);
69 
70  x = offset % TERMINAL_WIDTH;
71  y = offset / TERMINAL_WIDTH;
72 }
