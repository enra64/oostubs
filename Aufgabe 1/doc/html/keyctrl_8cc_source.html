52 bool Keyboard_Controller::key_decoded () {
53  bool done = false;
54 
55  // keys of MF II keyboards start sending one of two prefix bytes
56  if (code == prefix1 || code == prefix2) {
57  prefix = code;
58  return false;
59  }
60 
61  // releasing a key is important for the modifier keys (SHIFT, CTRL, ALT, ...)
62  // break codes from other keys are ignored
63  if (code & break_bit) {
64  // break code of a key is equal to the make code but additionaly the
65  // break_bit is set
66  code &= ~break_bit;
67  switch (code) {
68  case 42:
69  case 54:
70  gather.shift (false);
71  break;
72  case 56:
73  if (prefix == prefix1) {
74  gather.alt_right (false);
75  } else {
76  gather.alt_left (false);
77  }
78  break;
79  case 29:
80  if (prefix == prefix1) {
81  gather.ctrl_right (false);
82  } else {
83  gather.ctrl_left (false);
84  }
85  break;
86  }
87 
88  // a prefix is only valid for the following code
89  prefix = 0;
90 
91  // returns false since the break codes gives no additional information
92  return false;
93  }
94 
95  // a key has been pressed, if it was a modifier key like SHIFT, ALT, NUM_LOCK
96  // eg. only the internal state is changed. Return value 'false' shows that
97  // the input is not finished yet. If other keys have been pressed the ASCII
98  // and Scancode is read and 'ture' is returned.
99  switch (code) {
100  case 42:
101  case 54:
102  gather.shift (true);
103  break;
104  case 56:
105  if (prefix == prefix1) {
106  gather.alt_right (true);
107  } else {
108  gather.alt_left (true);
109  }
110  break;
111  case 29:
112  if (prefix == prefix1) {
113  gather.ctrl_right (true);
114  } else {
115  gather.ctrl_left (true);
116  }
117  break;
118  case 58:
119  gather.caps_lock (!gather.caps_lock ());
120  set_led (caps_lock, gather.caps_lock());
121  break;
122  case 70:
123  gather.scroll_lock (!gather.scroll_lock ());
124  set_led (scroll_lock, gather.scroll_lock());
125  break;
126  case 69: // NUM_LOCK or BREAK ?
127  if (gather.ctrl_left ()) { // BREAK
128  get_ascii_code ();
129  done = true;
130  } else { // NUM_LOCK
131  gather.num_lock (!gather.num_lock());
132  set_led (num_lock, gather.num_lock ());
133  }
134  break;
135  default: // other keys
136  // get ASCII codes from table, DONE
137  get_ascii_code ();
138  done = true;
139  }
140 
141  // a prefix is only valid for the following code
142  prefix = 0;
143 
144  if (done) {
145  return true;
146  } else {
147  return false;
148  }
149 }

154 void Keyboard_Controller::get_ascii_code () {
155  if (code == 53 && prefix == prefix1) { // special case scancode 53
156  gather.ascii ('/');
157  gather.scancode (Key::scan::div);
158  } else if (gather.num_lock () && !prefix && code>=71 && code<=83) {
159  gather.ascii (asc_num_tab[code-71]);
160  gather.scancode (scan_num_tab[code-71]);
161  } else if (gather.alt_right ()) {
162  gather.ascii (alt_tab[code]);
163  gather.scancode (code);
164  } else if (gather.shift ()) {
165  gather.ascii (shift_tab[code]);
166  gather.scancode (code);
167  } else if (gather.caps_lock ()) {
168  //Linux CapsLock, nur Buchstaben gross
169  if ((code>=16 && code<=26) || (code>=30 && code<=40) || (code>=44 && code<=50)) {
170  gather.ascii (shift_tab[code]);
171  gather.scancode (code);
172  } else {
173  gather.ascii (normal_tab[code]);
174  gather.scancode (code);
175  }
176  } else {
177  gather.ascii (normal_tab[code]);
178  gather.scancode (code);
179  }
180 }

235 void Keyboard_Controller::set_repeat_rate (unsigned char speed, unsigned char delay) {
236 
237  unsigned char status, reply;
238 
239  IntLock lock;
240 
241  data_port.outb (cmd_set_speed); // send command to keyboard
242  do {
243  status = ctrl_port.inb (); // wait for reply
244  } while ((status & outb) == 0);
245  reply = data_port.inb (); // get reply from controller
246  if (reply == ack) { // ok
247  data_port.outb (((delay & 3) << 5) | (speed & 31)); // set parameter
248  do {
249  status = ctrl_port.inb (); // wait for reply
250  } while ((status & outb) == 0);
251  reply = data_port.inb (); // get reply from controller
252  }
253 }

255 void Keyboard_Controller::set_led (Leds led, bool on) {
256 
257  unsigned char status, reply;
258 
259  IntLock lock;
260 
261  data_port.outb (cmd_set_led); // send command to keyboard
262  do {
263  status = ctrl_port.inb (); // wait for reply
264  } while ((status & outb) == 0);
265  reply = data_port.inb (); // get reply from controller
266  if (reply == ack) { // ok
267  if (on) {
268  leds |= led;
269  } else {
270  leds &= ~led;
271  }
272  data_port.outb (leds); // set parameter
273  do {
274  status = ctrl_port.inb (); // wait for reply
275  } while ((status & outb) == 0);
276  reply = data_port.inb (); // get reply from controller
277  }
