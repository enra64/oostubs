22 PIC::PIC() {
23  IO_Port ctrl_1(0x20), ctrl_2(0xa0), mask_1(0x21), mask_2(0xa1);
24  // http://wiki.osdev.org/8259_PIC#Initialisation
25  //
26  // When you enter protected mode (or even before hand, if you're not using GRUB) the
27  // first command you will need to give the two PICs is the initialise command (code 0x11).
28  // This command makes the PIC wait for 3 extra "initialisation words" on the data port.
29  // These bytes give the PIC:
30  // - Its vector offset. (ICW2)
31  // - Tell it how it is wired to master/slaves. (ICW3)
32  // - Gives additional information about the environment. (ICW4)
33 
34  ctrl_1.outb(0x11); // starts the initialization sequence (in cascade mode)
35  ctrl_2.outb(0x11);
36 
37  mask_1.outb(32); // ICW2: Master PIC vector offset
38  mask_2.outb(40); // ICW2: Slave PIC vector offset
39 
40  mask_1.outb(4); // ICW3: tell Master PIC that there is a slave PIC at IRQ2 (0000 0100)
41  mask_2.outb(2); // ICW3: tell Slave PIC its cascade identity (0000 0010)
42 
43  mask_1.outb(3); // ICW4: tell Master PIC that we are in 8086/88 Mode (bit 0 == 1) and Automatic EOI (bit 1 == 1)
44  mask_2.outb(3); // ICW4: tell Master PIC that we are in 8086/88 Mode (bit 0 == 1) and Automatic EOI (bit 1 == 1)
45 
46  // save the masks in ram to avoid IO access
47  mMaskMaster = 0xFB;
48  mMaskSlave = 0xFF;
49 
50  mask_1.outb(mMaskMaster); // Tell the Master PIC to ignore every interrupt except the interrupt connected to PIC2
51  mask_2.outb(mMaskSlave); // Tell the Slave PIC to ignore every interrupt
52

54  // set global interrupt enable flag
55  cpu.enable_int();
56 }

65 void PIC::allow(Interrupts interrupt){
66  // the first eight interrupts are on the master pic
67  bool isMaster = interrupt < 40;
68 
69  // get appropriate io_port
70  IO_Port maskPort(isMaster ? 0x21 : 0xa1);
71 
72  if(isMaster)
73  mMaskMaster &= ~(1 << (interrupt - 32));
74  else
75  mMaskSlave &= ~(1 << (interrupt - 40));
76 
77  // update pic masks
78  maskPort.outb(isMaster ? mMaskMaster : mMaskSlave);
