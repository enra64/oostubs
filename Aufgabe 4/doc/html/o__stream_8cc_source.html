70 O_Stream& O_Stream::operator << (unsigned long value) {
71  // allocate buffer to build the string of the number
72  char buf[32];
73  int i = 0;
74  int digit;
75  do {
76  // cut digit from value and advance to next digit
77  digit = value % base;
78  value /= base;
79  // put the corresponding character in the buffer
80  if(digit < 10) {
81  buf[i++] = digit + '0';
82  } else {
83  buf[i++] = digit - 10 + 'A';
84  }
85  // repeat while not all digits are processed
86  } while (value > 0);
87 
88  // put prefix to indicate base
89  switch (base) {
90  case 2: // 0b...
91  put('0');
92  put('b');
93  break;
94  case 8: // 0...
95  put('0');
96  break;
97  case 10: // no prefix
98  break;
99  case 16: //0x...
100  put('0');
101  put('x');
102  break;
103  }
104 
105  // write the string of the number
106  while(i>0) {
107  put(buf[--i]);
108  }
109  return *this;
110 }
