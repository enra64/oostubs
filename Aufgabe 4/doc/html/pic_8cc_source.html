21 PIC::PIC() : ctrl_1(0x20), ctrl_2(0xa0), mask_1(0x21), mask_2(0xa1) {
22  // http://wiki.osdev.org/8259_PIC#Initialisation
23  //
24  // When you enter protected mode (or even before hand, if you're not using GRUB) the
25  // first command you will need to give the two PICs is the initialise command (code 0x11).
26  // This command makes the PIC wait for 3 extra "initialisation words" on the data port.
27  // These bytes give the PIC:
28  // - Its vector offset. (ICW2)
29  // - Tell it how it is wired to master/slaves. (ICW3)
30  // - Gives additional information about the environment. (ICW4)
31 
32  ctrl_1.outb(0x11); // starts the initialization sequence (in cascade mode)
33  ctrl_2.outb(0x11);
34 
35  mask_1.outb(32); // ICW2: Master PIC vector offset
36  mask_2.outb(40); // ICW2: Slave PIC vector offset
37 
38  mask_1.outb(4); // ICW3: tell Master PIC that there is a slave PIC at IRQ2 (0000 0100)
39  mask_2.outb(2); // ICW3: tell Slave PIC its cascade identity (0000 0010)
40 
41  mask_1.outb(3); // ICW4: tell Master PIC that we are in 8086/88 Mode (bit 0 == 1) and Automatic EOI (bit 1 == 1)
42  mask_2.outb(3); // ICW4: tell Master PIC that we are in 8086/88 Mode (bit 0 == 1) and Automatic EOI (bit 1 == 1)
43 
44  mask_1.outb(0xFB); // Tell the Master PIC to ignore every interrupt except the interrupt connected to PIC2
45  mask_2.outb(0xFF); // Tell the Slave PIC to ignore every interrupt
46 
47  // globally enable interrupts
48  cpu.enable_int();
49 }

56 void PIC::allow(Interrupts interrupt){
57  // interrupt offset
58  int i = interrupt - timer;
59 
60  // determine which the responsible pic
61  IO_Port *port;
62  if( i < 8 ) {
63  port = &mask_1;
64  } else {
65  i -= 8;
66  port = &mask_2;
67  }
68 
69  // get mask from pic
70  unsigned char mask = port->inb();
71 
72  // allow interrupt by deleting the corresponding bit in the mask
73  mask &= 0xff ^ (1 << i);
74 
75  // apply updated mask
76  port->outb(mask);
77 }
